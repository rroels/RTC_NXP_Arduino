163void PCF2131_base::int_clear( uint8_t* rv )
164{
165 _reg_r( Control_2, rv, 3 );
166
167 if ( rv[ 0 ] & 0x90 ) // if interrupt flag set in Control_2
168 _reg_w( Control_2, rv[ 0 ] & ~((rv[ 0 ] & 0x90) | 0x49) ); // datasheet 7.11.5
169
170 if ( rv[ 1 ] & 0x08 ) // if interrupt flag set in Control_3
171 _reg_w( Control_3, rv[ 1 ] & ~(0x08) );
172
173 if ( rv[ 2 ] & 0xF0 ) // if interrupt flag set in Control_4
174 _reg_w( Control_4, rv[ 2 ] & ~(rv[ 2 ] & 0xF0) );
175}

177void PCF2131_base::periodic_interrupt_enable( periodic_int_select sel, int int_sel )
178{
179 if ( !sel ) {
180 _bit_op8( Control_1, ~0x03, 0x00 );
181 _bit_op8( int_mask_reg[ int_sel ][ 0 ], ~0x30, 0x30 );
182 return;
183 }
184
185 uint8_t v = (sel == EVERY_MINUTE) ? 0x02 : 0x01;
186
187 v=3;
188 _bit_op8( Control_1, ~0x03, v );
189 _bit_op8( int_mask_reg[ int_sel ][ 0 ], ~0x30, ~(v << 4) );
190}
